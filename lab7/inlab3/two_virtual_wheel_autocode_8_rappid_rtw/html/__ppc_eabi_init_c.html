   11      Use this file for C or C++.  Contains calls to initialize memory pools

   13      project does not use exceptions, then the initialization code for them
   14      is stripped out.  If they are used, the linker will add them

   16      with __ppc_eabi_init.c.
   17   
   18      Interface for board-level initialization and user-level initialization.
   19   
   20      If hardware initialization and pre-main user initialization are required,
   21      copy this file to your project directory and customize it (instead of
   22      customizing __start.c).
   23   
   24      Note that __init_hardware should not write on the stack until the
   25      memory controller is properly configured.
   26   
   27

   29   
   30      Initialize the hardware, including the memory controller.
   31   
   32      void __init_user(void)
   33   
   34      Allow the user to perform initialization before calling main().
   35   
   36      void __init_cpp(void)
   37

   39      constructors of which the cpp exception handling initialization
   40      is the first constructor, if exceptions are used.
   41

   47   
   48      This function simply stalls the debugger.  You may want to rewrite this
   49      function if you are using an OS.
   50   
   51      abort and exit
   52   
   53      In order to correctly implement the required startup/termination sequence for
   54      C and C++ programs, we need to have an exit() routine that can be called by
   55      the program startup code. The exit() routine is supposed to
   56   
   57      (1)	call any functions registered via atexit()
   58      (2) call destructors for any global objects
   59      (3)	flush any unwritten buffers, close any open files, etc.
   60      (4) terminates the program
   61   
   62      We don't, however, want to require the ANSI C library for every CodeWarrior
   63      program, since it drags in lots of code that may not be needed.
   64   
   65      Instead we provide a dummy exit() function which simply calls the destructors
   66      and terminates the program. We assume that any program which uses atexit()
   67      or <stdio.h> and which requires those cleanup behaviors will have linked with
   68      the ANSI C library, whose definition of exit() will override the one here.
   69   
   70      We similarly define a dummy abort() function (which is called by the default
   71      terminate() handler).
   72   
   73      Programs which rely on the proper ANSI C/C++ behavior must use the ANSI C
   74      library, and order it in the CodeWarrior project or command-line so that

  366   /***************************************************************************
  367    *** Requirements for SDA PIC/PID ***
  368   
  369      The basic requirement for SDA PIC/PID is that the startup code (.init) and
  370      the segments .sdata and .sdata2 all maintain their link time address
  371      relationship at runtime.  In other words if the link time addresses are:
  372      .init = 0x00002000
  373      .sdata2 = 0x00003000
  374      .sdata = 0x00004000
  375      but that .init somehow is executed at 0x00002500, then those link time
  376      addresses must all increment by 0x00000500 for their runtime addresses.
  377      Any segment that does not maintain the address relationship at runtime is
  378      considered external and must be addressed with absolute addresses.
  379      Segments that do maintain their link time address relationship at runtime
  380      are considered internal and must be addressed with pc and sda relative
  381      addressing.  More on external and internal segments, below.
  382   
  383      Having .init, .sdata2 and .sdata, as well as other internal segments, end
  384      up at runtime without the link time relationship is not detected in this
  385      runtime but it is at least technically possible for someone to modify the
  386      runtime to support this, at least for a specific application.
  387   
  388    *** Internal and External segments and references ***
  389   
  390      At link time, the linker determines whether code and data segments are
  391      external or internal.  Internal segments reference their data as far or
  392      near offsets of the small data registers r2 and r13.  Their code
  393      references are normally pc relative, but if far code references are
  394      required, they also use offsets of the small data registers.  Internal
  395      segments can also reference code and data in other internal segments
  396      with the same addressing that they would use for their own code and data.
  397      By default, all segments in your application are considered internal with
  398      one exception listed below.
  399   
  400      External segments are segments that are at absolute addresses.
  401      Segments with a name like .abs.xxxxxxxx, where xxxxxxxx is a hex address
  402      are considered external.  External segments reference their data with
  403      absolute addressing and code references within the segment can be pc
  404      relative or absolute.  Any other segment must use absolute references
  405      to reference code or data in external segments.  External segments must
  406      reference an internal segment with small data registers for code and data.
  407   
  408      Akin to external segments are external symbol references.  These are
  409      symbols, usually linker generated symbols, that are determined to be not
  410      within any segment in your application.  They will be referenced with
  411      absolute addressing.  All symbols in an external segment are considered
  412      to be external symbol references.
  413   
  414      The ldf has 2 mechanisms for you to override the linker's defaults:
  415      1.  The MEMORY directive now supports
  416      addr_mode = external | internal // default is internal
  417      addr_mode is ignored unless you have chosen the SDA PIC/PID ABI in the EPPC
  418      Target Pref panel.
  419   
  420      2. New directives INTERNAL_SYMBOL and EXTERNAL_SYMBOL can be used to force
  421      the addressing of global symbols.  These directives are of the
  422      form XXXL_SYMBOL { sym1, sym2, symN } where the symbols are the link time
  423      symbol names (mangled for c++).
  424   
  425      The auto-generated linker defined start and end symbols for loadable
  426      segments are internal if they are addresses into internal segments and
  427      external if they are for external segments.  All other linker defined
  428      symbols you create in a lcf are considered external unless you redefine
  429      them with INTERNAL_SYMBOL.  The linker also defines some linker defined
  430      symbols for its own use and they have the following default settings:
  431   
  432      _stack_addr [top of the stack - comes from pref panel settings] external
  433      _stack_end [bottom of the stack - comes from pref panel settings] external
  434      _heap_addr [bottom of the heap - comes from pref panel settings] external
  435      _heap_end [top of the heap - comes from pref panel settings] external
  436    // the next two need to be treated as internal to be EABI compliant so
  437    // you may not redefine them.
  438      _SDA_BASE_ [.sdata + 0x00008000] internal
  439      _SDA2_BASE_ [.sdata2 + 0x00008000] internal
  440    // The previous small data pointers rely on the small data registers being
  441    // properly initialized before being accessible.  However, it is hard to
  442    // initialize those pointers without accessing them as absolutes.
  443    // The following two symbols are external versions of _SDA_BASE_ and
  444    // _SDA2_BASE_ which can be used as absolutes.  You may not redefine them.
  445      _ABS_SDA_BASE_ [.sdata + 0x00008000] external
  446      _ABS_SDA2_BASE_ [.sdata2 + 0x00008000] external
  447    // the following are numbers, not addresses and they must be treated
  448    // as external.  You may not redefine them.
  449      _nbfunctions [number of functions in program - deprecated] external
  450      SIZEOF_HEADERS [size of the segment headers in a linux app] external
  451   
  452      Especially note that the stack and heap linker generated symbols are
  453      external.  It may be more practical in a SDA PIC/PID application to
  454      make the heap and stack be contiguous with an internal segment and
  455      define them as internal.
  456   
  457    *** Uses for SDA PIC/PID ***
  458   
  459      The runtime was developed for 3 different situations:
  460   
  461      1. All code and data segments are internal.  The simplest case would be
  462      for all segments to use the same MEMORY directive and to have all of the
  463      bss type segments at the end.  In such a simple case, the application could
  464      be converted to a binary file and linked into another application which
  465      could copy it to ram and jump to its entry point.
  466   
  467      2. All of the essential segments are internal and therefore moveable but
  468      there may be some external segments which are absolute.  This situation is
  469      probably difficult to test but you can download the entire application
  470      to the chip and at least debug it at its link time addresses.
  471   
  472      3. Like 2, there are internal and external segments but the application is
  473      linked as a ROM image - the application does not need to be flashed to
  474      ROM, however.  It is possible to change the ROM Image Address to be an
  475      address into RAM and have the debugger download the image to the RAM
  476      address.  Alternatively, you could have the ROM image converted to a bin
  477      file and link it into another application as in 1, above.  The structures
  478      used in __init_data(), _rom_copy_info and __bss_init_info, have been
  479      modified for SDA PIC/PID to have an extra field which tells the runtime
  480      where the segment is internal or external so that the internal segments
  481      will be copied to position relative addresses and the external segments
  482      will be copied to absolute addresses.
  483   
  484    *** Compiler/Linker support for SDA Addressing ***
  485      You build a SDA PIC/PID application by selecting SDA PIC/PID in the ABI
  486      popup in the EPPC Target Preference panel.  When you chose that, the
  487      compiler will define a simple that you can use to guard PIC/PID source.
  488   
  489      #if __option(sda_pic_pid) // is true if you have chosen SDA PIC/PID ABI
  490   
  491      If you then link, the linker will generate a table for the runtime files
  492      __ppc_eabi_init.cpp/__ppc_eabi_init.c to use.  If your application has
  493      absolute addressing relocations (which it would by default) you will
  494      probably get linker warnings telling you that those relocations may
  495      cause a problem.  If that happens, you have two choices.  You can change
  496      the Code Model popup in the EPPC Target Preference panel to be
  497      SDA Based PIC/PID Addressing for all of your sources and libraries, and/or
  498      check the Tune Relocations checkbox in the EPPC Target Preference panel.
  499      This new option is only available for the EABI and SDA PIC/PID ABIs.  For
  500      EABI it changes 14 bit branch relocations to 24 bit branch relocations
  501      only if they can not reach the calling site from the original relocation.
  502      For SDA PIC/PID, it changes absolute addressed references of data from
  503      code to use a small data register instead of r0 and changes absolute
  504      code to code references to use the pc relative relocations.  In general,
  505      this will work well for normal compiled code.  It is always possible
  506      to link in an assembly file that does not behave in a standard way.
  507   
  508      Whether you use the Code Model SDA Based PIC/PID Addressing or not,
  509      #pragma section has been modified to accept far_sda_rel for the data_mode
  510      and code_mode options.  If you leave those options out, the compiler
  511      will use the Code Model to determine the appropriate modes.
  512   
  513      Absolute Addressing defaults to data_mode = far_abs and code_mode = pc_rel.
  514      SDA Based PIC/PID Addressing defaults to data_mode = far_sda_rel and
  515      code_mode = pc_rel.

  520   /*
  521    *	__get_runtime_linktime_delta
  522    *
  523    *	Assumes that r13 and r2 are initialized correctly and that they always
  524    *	stay in the same relationship to each other.
  525    *
  526    *	Utility function to get difference between runtime and linktime addresses.
  527    *
  528    */
  529   /***************************************************************************/
  530   static __asm int __get_runtime_linktime_delta(void)
  531   {
  532     nofralloc
  533       /* _ABS_SDA_BASE_, generated by the linker, is the same value as 		 	    */
  534       /* _SDA_BASE_ but forces the linker to treat it as absolutetly addressed 	*/
  535       /* and not to substitute r13 for r0.  We therefore end up with r3 exactly */
  536       /* as calculated at linktime. 												                    */
  537       addis r3, r0, _ABS_SDA_BASE_@ha
  538       addi r3, r3, _ABS_SDA_BASE_@l
  539       /* Subtract the runtime from the linktime address.							          */
  540       sub r3,r13,r3
  541       blr
  542   }                                      /***************************************************************************/
  543   /*
  544    *	__mwerks_fixup_relocations
  545    *
  546    *	Modify initialized pointers in the data sections to reflect correct
  547    *	runtime addresses.  The linker created a table, _f_mwerks_fixup, of
  548    *	offsets into the data sections that need modification.  This function
  549    *	loads the pointer at that offset and applies the runtime/linktime
  550    *	delta.  We wait to call this function until after __init_data.
  551    *
  552    */
  553   /***************************************************************************/
  554   static __asm void __mwerks_fixup_relocations(void)
  555   {
  556     nofralloc
  557       /* r13, r2, and r1 have their correct runtime values.						*/
  558       mflr r0
  559       /* If runtime and linktime addresses are the same, no fixup is necessary.	*/
  560       bl __get_runtime_linktime_delta
  561       cmpli cr0,r3,0
  562       beq no_fixup
  563       /* Initialize runtime addresses of the fixup table.  _f_mwerks_fixup is the	*/
  564       /* beginning of the table and _e_mwerks_fixup is the end.  Note that r0 	  */
  565       /* will get converted to r13 or r2 by the linker. 							            */
  566       addis r5,r0,_f_mwerks_fixup@ha
  567       addis r4,r0,_e_mwerks_fixup@ha
  568       addi r5,r5,_f_mwerks_fixup@l
  569       addi r4,r4,_e_mwerks_fixup@l
  570       /* Compute number of entries ((_e_mwerks_fixup - _f_mwerks_fixup) / 4).		*/
  571       subf r6,r5,r4
  572       /* If there are no entries, bail.	*/
  573       cmpli cr0,r6,0
  574       beq no_fixup
  575       srwi r6,r6,2
  576       mtctr r6
  577       /* We use pre-increment, so move to first real entry - 4.	*/
  578       subi r5,r5,4
  579       fixup_loop:
  580       /* Load address of relocation. */
  581       lwzu r6,4(r5)
  582     /* Update address using SDA delta. */
  583       add r6,r6,r3
  584       /* Load value at relocation. */
  585       lwz r7,0(r6)
  586     /* Update value using SDA delta. */
  587       add r7,r7,r3
  588       /* Store updated value.	*/
  589       stw r7,0(r6)
  590       bdnz fixup_loop
  591       no_fixup:
  592       mtlr r0
  593       blr
  594   }                                      /***************************************************************************/

  609     nofralloc                            /* see above on usage of nofralloc */
  610       mflr r0
  611       /* _ABS_SDA_BASE_, generated by the linker, is the same value as 		 	      */
  612       /* _SDA_BASE_ but forces the linker to treat it as absolutetly addressed 	  */
  613       /* and not to substitute r13 for r0.  We therefore end up with r13 exactly 	*/
  614       /* as calculated at linktime. 												                      */
  615       addis r13, r0, _ABS_SDA_BASE_@ha
  616       addi r13, r13, _ABS_SDA_BASE_@l
  617       /* _ABS_SDA2_BASE_, generated by the linker, is the same value as 		 	  */
  618       /* _SDA2_BASE_ but forces the linker to treat it as absolutetly addressed */
  619       /* and not to substitute r2 for r0.  We therefore end up with r2 exactly 	*/
  620       /* as calculated at linktime. 												                    */
  621       addis r2, r0, _ABS_SDA2_BASE_@ha
  622       addi r2, r2, _ABS_SDA2_BASE_@l
  623       /* Grab the runtime address of __get_linktime_address.  The bcl will put	*/
  624       /* __get_linktime_address into LR. */
  625       bcl 20,31,__get_linktime_address
  626       entry __get_linktime_address
  627       /* Move runtime address of __get_linktime_address from LR to r4. */
  628       mflr r4
  629       /* Grab the linktime address of __get_linktime_address.  Note that r0 will	*/
  630       /* get converted to r13 or r2 by the linker but because r13 and r2 may not	*/
  631       /* reflect their runtime values yet, r3 might not equal r4.				          */
  632       addis r3,r0,__get_linktime_address@ha
  633       addi r3,r3,__get_linktime_address@l
  634       /* Subtract the runtime from the linktime address.							*/
  635       sub r3,r4,r3
  636       /* If runtime and linktime addresses are the same, no fixup is necessary.	*/
  637       cmpli cr0,r3,0
  638       beq no_fixup
  639       /* Add the delta into small data pointers so they will reflect the current 	*/
  640       /* position of code and data.											            	            */
  641       add r13,r13,r3
  642       add r2,r2,r3
  643       no_fixup:
  644       /* Initialize runtime address of _stack_addr.  Note that r0 will			*/
  645       /* get converted to r13 or r2 by the linker if _stack_addr is 				*/
  646       /* treated as an internal symbol (default is external).						    */
  647       addis r1, r0, _stack_addr@ha
  648       addi r1, r1, _stack_addr@l
  649       mtlr r0
  650       blr
  651   }                                      /***************************************************************************/
  652   /*
  653    *	__init_data
  654    *
  655    *	Initialize all (RAM) data sections, copying ROM sections as necessary.
  656    *
  657    *	dst			destination RAM address
  658    *	size		number of bytes to zero
  659    *
  660    *	we now overload the function in the event that the user wants pic pid
  661    *
  662    */
  663   /***************************************************************************/
  664   extern void __init_data(void)
  665   {
  666     __pic_rom_copy_info *dci;
  667     __pic_bss_init_info *bii;
  668     int delta = __get_runtime_linktime_delta();
  669     unsigned int size;
  670     unsigned char *ram_addr;             /* address in ram (executing address) */
  671     unsigned char *ram_end;              /* end address in ram (executing address) */
  672     unsigned char *rom_addr;             /* address in rom */
  673     unsigned char *rom_end;              /* end address in rom */
  674     int internal;
  675   
  676     /* for sda pic pid, if ram_end and ram_addr are swapped so that ram_end < ram_addr, 	 	  */
  677     /* then segment is internal, otherwise it is external.  for eabi, addresses are external 	*/
  678     dci = _pic_rom_copy_info;
  679     while (1) {
  680       if (dci->rom_addr == 0 && dci->ram_addr == 0 && dci->ram_end == 0)
  681         break;
  682       internal = (dci->rom_end < dci->rom_addr);
  683       if (internal) {
  684         rom_addr = dci->rom_end + delta;
  685         rom_end = dci->rom_addr + delta;
  686       } else {
  687         rom_addr = dci->rom_addr;
  688         rom_end = dci->rom_end;
  689       }
  690   
  691       internal = (dci->ram_end < dci->ram_addr);
  692       if (internal) {
  693         ram_addr = dci->ram_end + delta;
  694         ram_end = dci->ram_addr + delta;
  695       } else {
  696         ram_addr = dci->ram_addr;
  697         ram_end = dci->ram_end;
  698       }
  699   
  700       size = (unsigned int) (ram_end - ram_addr);
  701       __copy_rom_section(ram_addr, rom_addr, size);
  702       dci++;
  703     }
  704   
  705     /* Initialize with zeros: */
  706     bii = _pic_bss_init_info;
  707     while (1) {
  708       if (bii->ram_addr == 0 && bii->ram_end == 0)
  709         break;
  710       internal = (bii->ram_end < bii->ram_addr);
  711       if (internal) {
  712         ram_addr = bii->ram_end + delta;
  713         ram_end = bii->ram_addr + delta;
  714       } else {
  715         ram_addr = bii->ram_addr;
  716         ram_end = bii->ram_end;
  717       }
  718   
  719       size = (unsigned int) (ram_end - ram_addr);
  720       __init_bss_section(ram_addr, size);
  721       bii++;
  722     }
  723   
  724     /* now that data is copied to ram, we can fix up initialized pointers */
  725     __mwerks_fixup_relocations();
  726   }

  730    *	__exception_info_constants
  731    *
  732    *	Called from __init_cpp_exceptions.cpp
  733    *
  734    *	info		address of linker generated exception table info
  735    *	R2			address of variable to hold the value in r2
  736    *
  737    *	we now overload the function in the event that the user wants pic pid

  752   /*
  753    *	__find_exception_addresses
  754    *
  755    *	Called from ExceptionPPC.cp
  756    *
  757    *	info		address of linker generated exception table info
  758    *	returnaddr	address within a code segment
  759    *	ex_start	address of variable to hold address of exception table info
  760    *	ex_end		address of variable to hold end address of exception table info
  761    *
  762    *	we now overload the function in the event that the user wants pic pid
  763    *
  764    */
  765   /***************************************************************************/
  766   extern int __find_exception_addresses(void *info, char *returnaddr, void
  767     **ex_start, void **ex_end)
  768   {
  769     __pic_eti_init_info *eti_info = (__pic_eti_init_info*)info;
  770     unsigned char *start;                /* address */
  771     unsigned char *end;                  /* end address */
  772     int internal;
  773     int delta = __get_runtime_linktime_delta();
  774     while (1) {
  775       if (eti_info->code_end - eti_info->code_start == 0)
  776         break;
  777       internal = (eti_info->code_end < eti_info->code_start);
  778       if (internal) {
  779         start = eti_info->code_end + delta;
  780         end = eti_info->code_start + delta;
  781       } else {
  782         start = eti_info->code_start;
  783         end = eti_info->code_end;
  784       }
  785   
  786       if ((unsigned char*)returnaddr>=start && (unsigned char*)returnaddr<end) {
  787         internal = (eti_info->eti_end < eti_info->eti_start);
  788         if (internal) {
  789           start = eti_info->eti_end + delta;
  790           end = eti_info->eti_start + delta;
  791         } else {
  792           start = eti_info->eti_start;
  793           end = eti_info->eti_end;
  794         }
  795   
  796         *ex_start = start;
  797         *ex_end = end;
  798         return(1);
  799       }
  800   
  801       eti_info++;
  802     }
  803   
  804     return(0);
  805   }
