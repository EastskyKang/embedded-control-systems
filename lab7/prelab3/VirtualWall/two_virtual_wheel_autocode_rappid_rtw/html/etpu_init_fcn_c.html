  120   int16_t etpu_init(void)
  121   {
  122     // ETPU Initialization Code
  123     int16_t error_code = 0;
  124     uint32_t chanConfigParam = 0;
  125   
  126     // Initialize eTPU host side
  127     error_code = fs_etpu_init(etpu_config_A, (uint32_t *)etpu_code,(uint32_t)
  128       ETPU_CODE_RAM_SIZE, (uint32_t *)etpu_globals, (uint32_t) ETPU_GLOBALS_SIZE,
  129       (uint8_t)load_microcode);
  130   
  131     // ********************************************************************
  132     // ***       Add eTPU Function Initialization calls here            ***
  133     // ********************************************************************
  134     fs_etpu_qd_init(3,
  135                     FS_ETPU_CHAN_NOT_USED,
  136                     FS_ETPU_CHAN_NOT_USED,
  137                     FS_ETPU_QD_PRIM_SEC,
  138                     FS_ETPU_PRIORITY_DISABLE,
  139                     FS_ETPU_QD_CONFIGURATION_0,
  140                     FS_ETPU_TCR1,
  141                     0,                   /* pc_max 0 - automatic position counter reset not performed */
  142                     0.40 * 1000,         /* slow_normal_threshold */
  143                     0.35 *1000,          /* normal_slow_threshold */
  144                     0.80 *1000,          /* normal_fast_threshold */
  145                     0.75 *1000,          /* fast_normal_threshold */
  146                     0x733333,            /* window_ratio1: 0.9*2^23 */
  147                     0x8CCCCC,            /* window_ratio2: 1.1*2^23 */
  148                     FS_ETPU_QD_HOME_TRANS_LOW_HIGH, /* detection of home_transition low to high */
  149                     FS_ETPU_QD_INDEX_PULSE_POSITIVE, /* index_pulse of positive polarity */
  150                     FS_ETPU_QD_INDEX_PC_NO_RESET, /* PC is NOT reset to pc_init on Index transition */
  151                     32000000,
  152                     4000);
  153   
  154     // ETPU A Channel 3 initialization, set to ETPU Function Quadrature Decoding
  155     chanConfigParam = (FS_ETPU_INTERRUPT_DISABLE | FS_ETPU_DMA_DISABLE |
  156                        FS_ETPU_OUTPUT_DISABLE_OFF | (0x0000 << 24 ) | (0x0002 <<
  157       28) );
  158     fs_etpu_chan_init( QD1, FS_ETPU_QD_FUNCTION_NUMBER, 0, FS_ETPU_QD_NUM_PARMS,
  159                       chanConfigParam, 0);
  160     fs_etpu_qd_set_pc_interrupts(3,0,0);
  161     fs_etpu_qd_enable_pc_interrupts(3);
  162     fs_etpu_qd_enable(3,FS_ETPU_CHAN_NOT_USED,FS_ETPU_CHAN_NOT_USED,
  163                       FS_ETPU_QD_PRIM_SEC,FS_ETPU_PRIORITY_HIGH);
  164     return(error_code);
  165   }
