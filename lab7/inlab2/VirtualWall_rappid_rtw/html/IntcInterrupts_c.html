   71   #pragma interrupt on
   72   void INTC_INTCInterruptHandler(void)
   73   {
   74       INTCInterruptFn *handlerFn = (INTCInterruptFn*)(*(unsigned int*)&INTC.IACKR.R);
   75   
   76       (**handlerFn)();
   77   
   78       INTC.EOIR.R = 0;
   79   }
   80   #pragma interrupt off

   84   __asm void INTC_INTCInterruptHandler(void)
   85   {
   86   nofralloc
   87   prologINTC:
   88       stwu    r1, -0x50 (r1)      /* Create stack frame */
   89       stw r0,  0x24 (r1)          /* Store r0 working register  */
   90   
   91       /* Save SRR0 and SRR1 */
   92       mfsrr1  r0                  /* Store SRR1 (must be done before enabling EE) */
   93       stw     r0,  0x10 (r1)
   94       mfsrr0  r0                  /* Store SRR0 (must be done before enabling EE) */
   95       stw     r0,  0x0C (r1)
   96   
   97       /* Clear request to processor; r3 contains the address of the ISR */
   98       stw     r3,  0x28 (r1)      /* Store r3 */
   99       lis     r3, INTC_IACKR@h    /* Read pointer into ISR Vector Table & store in r3 */
  100       ori     r3, r3, INTC_IACKR@l
  101       lwz     r3, 0x0(r3)         /* Load INTC_IACKR, which clears request to processor */
  102       lwz     r3, 0x0(r3)         /* Read ISR address from ISR Vector Table using pointer  */
  103   
  104       /* Enable processor recognition of interrupts */
  105       wrteei  1                   /* Set MSR[EE]=1  */
  106   
  107       /* Save rest of context required by EABI */
  108       stw     r12, 0x4C (r1)      /* Store r12 */
  109       stw     r11, 0x48 (r1)      /* Store r11 */
  110       stw     r10, 0x44 (r1)      /* Store r10 */
  111       stw     r9,  0x40 (r1)      /* Store r9 */
  112       stw     r8,  0x3C (r1)      /* Store r8 */
  113       stw     r7,  0x38 (r1)      /* Store r7 */
  114       stw     r6,  0x34 (r1)      /* Store r6 */
  115       stw     r5,  0x30 (r1)      /* Store r5 */
  116       stw     r4,  0x2C (r1)      /* Store r4 */
  117       mfcr    r0                  /* Store CR */
  118       stw     r0,  0x20 (r1)
  119       mfxer   r0                  /* Store XER */
  120       stw     r0,  0x1C (r1)
  121       mfctr   r0                  /* Store CTR */
  122       stw     r0,  0x18 (r1)
  123       mflr    r0                  /* Store LR */
  124       stw     r0,  0x14 (r1)
  125   
  126       /* Branch to ISR handler address from SW vector table */
  127       mtlr    r3                  /* Store ISR address to LR to use for branching later  */
  128       blrl                        /* Branch to ISR, but return here */
  129   
  130   epilogINTC:
  131       /* Restore context required by EABI (except working registers) */
  132       lwz     r0,  0x14 (r1)      /* Restore LR */
  133       mtlr    r0
  134       lwz     r0,  0x18 (r1)      /* Restore CTR */
  135       mtctr   r0
  136       lwz     r0,  0x1C (r1)      /* Restore XER */
  137       mtxer   r0
  138       lwz     r0,  0x20 (r1)      /* Restore CR */
  139       mtcrf   0xff, r0
  140       lwz     r5,  0x30 (r1)      /* Restore r5 */
  141       lwz     r6,  0x34 (r1)      /* Restore r6 */
  142       lwz     r7,  0x38 (r1)      /* Restore r7 */
  143       lwz     r8,  0x3C (r1)      /* Restore r8 */
  144       lwz     r9,  0x40 (r1)      /* Restore r9 */
  145       lwz     r10, 0x44 (r1)      /* Restore r10 */
  146       lwz     r11, 0x48 (r1)      /* Restore r11 */
  147       lwz     r12, 0x4C (r1)      /* Restore r12 */
  148   
  149       /* Disable processor recognition of interrupts */
  150       wrteei  0
  151   
  152       /* Ensure interrupt flag has finished clearing */
  153       mbar    0
  154   
  155       /* Write 0 to INTC_EOIR, informing INTC to lower priority */
  156       li      r3, 0
  157       lis     r4, INTC_EOIR@h     /* Load upper half of INTC_EOIR address to r4 */
  158       ori     r4, r4, INTC_EOIR@l
  159       stw     r3, 0(r4)           /* Write 0 to INTC_EOIR */
  160   
  161       /* Restore Working Registers */
  162       lwz     r3,  0x28 (r1)      /* Restore r3 */
  163       lwz     r4,  0x2C (r1)      /* Restore r4 */
  164   
  165       /* Retrieve SRR0 and SRR1 */
  166       lwz     r0,  0x0C (r1)      /* Restore SRR0 */
  167       mtsrr0  r0
  168       lwz     r0,  0x10 (r1)      /* Restore SRR1 */
  169       mtsrr1  r0
  170   
  171       /* Restore Other Working Registers */
  172       lwz     r0,  0x24 (r1)      /* Restore r0 */
  173   
  174       /* Restore space on stack */
  175       addi    r1, r1, 0x50
  176   
  177       /* End of Interrupt */
  178       rfi
  179   }
  180   #endif

  185   /**
  186    * This function can be used to install an interrupt handler for a given
  187    * interrupt vector. It will also set the Priority Status Register for the
  188    * source to the one given
  189    */
  190   void INTC_InstallINTCInterruptHandler(INTCInterruptFn handlerFn, unsigned short vectorNum,
  191                                         unsigned char psrPriority)
  192   {
  193       /* Set the function pointer in the ISR Handler table */
  194       INTCInterruptsHandlerTable[vectorNum] = handlerFn;
  195       /* Set the PSR Priority */
  196       INTC.PSR[vectorNum].B.PRI = psrPriority;
  197   }

  199   /**
  200    * This function will setup the PowerPC Zen core to jump to an Interrupt Service Routine
  201    * handler. This function can be used from user_init() (no stack frame, no memory access).
  202    */
  203   __asm void INTC_InitINTCInterrupts( void )
  204   {
  205   nofralloc
  206   
  207       mflr    r29
  208   
  209       /* IVOR4 is for external interrupts (special purpose register #404)
  210       * This code enables the ISR handler for external interrupt code to address
  211       * INTC_INTCInterruptHandler (incl. large address by setting the IVPR value).
  212       */
  213       lis     r0, INTC_INTCInterruptHandler@h
  214       ori     r0, r0, INTC_INTCInterruptHandler@l
  215       mtivor4 r0
  216   
  217       /* Enable external interrupts in the Machine State Register. Set MSR[EE]=1. */
  218       wrteei  1
  219   
  220       /* Set the location of the ISR Handler Table in INTC IACKR Register */
  221       lis     r0, INTCInterruptsHandlerTable@h
  222       ori     r0, r0, INTCInterruptsHandlerTable@l
  223       lis     r3,INTC_IACKR@h
  224       ori     r3,r3,INTC_IACKR@l
  225       stw     r0, 0(r3)
  226   
  227       mtlr    r29
  228   
  229       blr
  230   }
