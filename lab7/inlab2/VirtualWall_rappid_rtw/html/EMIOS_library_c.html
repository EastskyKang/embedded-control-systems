   30   /*********************************************************************************
   31      Function Name :emios_init_opwfm_fnc()
   32   
   33      Description:To Set the initial value sof A and B registers with duty cycle and period
   34   
   35      Inputs: emios Channel Number, Duty cycle, Period
   36   
   37      Outputs: None
   38   
   39    **********************************************************************************/
   40   /* set Init values for OPWFM */
   41   void emios_init_opwfm_fnc(uint16_t emios_chan, uint32_t dutyCycle, uint32_t
   42     period)
   43   {
   44     /* EMIOS Unified Channel Setup Mode : Pulse Width and Frequency Modulation Mode:
   45        In this mode, register A1 contains the duty cycle and register B1 contains the
   46        period of the output signal. */
   47     EMIOS.CH[emios_chan].CADR.R = dutyCycle;/* Channel A Data Register - CADR0   */
   48     EMIOS.CH[emios_chan].CBDR.R = period;/* Channel B Data Register - CBDR0   */
   49   }

   51   /*********************************************************************************
   52      Function Name :emios_gen_opwfm_fnc()
   53   
   54      Description:To generate Output PWM with duty cycle and period
   55   
   56      Inputs: emios Channel Number, Duty cycle, Period, emios Clock
   57   
   58      Outputs: None
   59   
   60    **********************************************************************************/
   61   void emios_gen_opwfm_fnc(uint16_t emios_chan, uint32_t dutyCycle, uint32_t
   62     frequency, uint32_t sysClock, uint16_t prescaler)
   63   {
   64     uint32_t A1;
   65     uint32_t B1;
   66     uint32_t currCounter;
   67     uint32_t buff = 30;                  /*Time in UC clock cycles required to complete FORCMA */
   68     if (dutyCycle > 99 ) {               /* Write out to A & B registers for case of 100 % dutycycle */
   69       EMIOS.CH[emios_chan].CADR.R = 0;   /* Write the Duty Cycle Value for the Channel */
   70       EMIOS.CH[emios_chan].CBDR.R = 0;   /* Write the Frequency Value for the Channel */
   71       EMIOS.CH[emios_chan].CCR.B.FORCMB = 1;
   72     } else if (dutyCycle < 1 ) {         /* Write out to A & B registers for case of 0 % dutycycle */
   73       EMIOS.CH[emios_chan].CADR.R = 100; /* Write the Duty Cycle Value for the Channel */
   74       EMIOS.CH[emios_chan].CBDR.R = 100; /* Write the Frequency Value for the Channel */
   75       EMIOS.CH[emios_chan].CCR.B.FORCMA = 1;
   76     } else {                             /* Write out to A & B registers for a dutycycle or frequency change */
   77       B1 = (sysClock*1000000)/(prescaler*frequency) ;
   78       A1 = ((dutyCycle*B1)/100) ;
   79       currCounter = EMIOS.CH[emios_chan].CCNTR.R ;
   80       if (A1 > 1)
   81         A1 = A1-1;                       /*Account for A reg value off by 1*/
   82       EMIOS.CH[emios_chan].CADR.R = A1;
   83       if (A1 < currCounter )             /*To prevent counter wrap condition*/
   84       {
   85         if (currCounter + buff > EMIOS.CH[emios_chan].CBDR.R ) {
   86         } else {
   87           EMIOS.CH[emios_chan].CCR.B.FORCMA = 1;
   88         }
   89       }
   90   
   91       if (B1 > 1)
   92         B1 = B1-1;                       /*Account for B reg value off by 1*/
   93       EMIOS.CH[emios_chan].CBDR.R = B1;
   94     }
   95   }

  159   /*********************************************************************************
  160      Function Name: emios_get_pulse_time_fnc()
  161   
  162      Description: Get the time required for a desired number of events
  163   
  164      Inputs: eMIOS Channel Number, max_counter, Max Counter value, System Clock, Channel Prescaler
  165   
  166      Outputs: Pulse Accumulated Time
  167   
  168    **********************************************************************************/
  169   uint32_t emios_get_pulse_time_fnc(uint16_t emios_chan,uint32_t counter_max,
  170     uint32_t sysClock, uint16_t prescaler)
  171   {
  172     int32_t counterDiff;
  173     uint32_t AReg;
  174     uint32_t BReg;
  175     uint32_t pulse_time;
  176     AReg = EMIOS.CH[emios_chan].CADR.R;
  177     BReg = EMIOS.CH[emios_chan].CBDR.R;
  178     counterDiff = (int32_t)(AReg - BReg);
  179     if (counterDiff < 0) {
  180       counterDiff = (int32_t)( counterDiff + counter_max);
  181     }
  182   
  183     /* calculate the Pulse Accumulation time in us */
  184     pulse_time = (uint32_t)((counterDiff*prescaler)/sysClock);
  185     return(pulse_time);
  186   }

  198   void emios_set_counter_fnc(uint16_t emios_chan, uint32_t counter_max)
  199   {
  200     EMIOS.CH[emios_chan].CADR.R = counter_max;/* Channel A Data Register - CADR0   */
  201     EMIOS.CH[emios_chan].CBDR.R = 0x00000000;/* Channel  B Data Register - CBDR0   */
  202     EMIOS.CH[emios_chan].CCR.R = 0x0C000610;/* Channel  Control Register - CCR0 */

  205   /*********************************************************************************
  206      Function Name :emios_get_ipwm_fnc()
  207   
  208      Description:To measure the width of the Input Pulse
  209   
  210      Inputs: emios Channel Number,Max Counter value, System Clock, Channel Prescaler
  211   
  212      Outputs: Return Period of the width Pulse wave in us
  213   
  214    **********************************************************************************/
  215   uint32_t emios_get_ipwm_fnc(uint16_t emios_chan,uint32_t counter_max, uint32_t
  216     sysClock, uint16_t prescaler)
  217   {
  218     int32_t counterDiff;
  219     uint32_t AReg;
  220     uint32_t BReg;
  221     uint32_t pWidth;
  222     AReg = EMIOS.CH[emios_chan].CADR.R;
  223     BReg = EMIOS.CH[emios_chan].CBDR.R;
  224     counterDiff = (int32_t)(AReg - BReg);
  225     if (counterDiff < 0)                 /* Errata condition rollover occured,correct pulse width */
  226     {
  227       /*Add Counter Maximum for the channelto the value before rollover */
  228       counterDiff = (int32_t)(counterDiff + counter_max);
  229     }
  230   
  231     /*calculate the Pulse width in us */
  232     pWidth = (uint32_t)((counterDiff*prescaler)/sysClock);
  233     return(pWidth);
  234   }
