   28     /* Symbol L2SRAM_LOCATION is defined in the application linker command file (.lcf)
   29        It is defined to the start of the L2SRAM of the MPC5554.
   30      */
   31     /*lint -esym(752, L2SRAM_LOCATION) */
   32     extern long L2SRAM_LOCATION;
   33     __asm void INIT_Derivative(void)
   34     {
   35       nofralloc
   36         /* MPC5554 L2SRAM initialization code                         */
   37         lis r11,L2SRAM_LOCATION@h        /* Base address of the L2SRAM, 64-bit word aligned */
   38         ori r11,r11,L2SRAM_LOCATION@l
   39         li r12,512                       /* Loop counter to get all of L2SRAM; 64k/4 bytes/32 GPRs = 512 */
   40         mtctr r12
   41         init_l2sram_loop:
   42         stmw r0,0(r11)                   /* Write all 32 GPRs to L2SRAM */
   43         addi r11,r11,128                 /* Inc the ram ptr; 32 GPRs * 4 bytes = 128 */
   44         bdnz init_l2sram_loop            /* Loop for 64k of L2SRAM */
   45         blr
   46     }                                    /*******************************************************/

  100     /*----------------------------------------------------------------------------*/
  101     /* SIU External Bus Interface                                                 */
  102     /*----------------------------------------------------------------------------*/
  103   
  104     /* Initialize a set of contiguous PCRs:               */
  105     /* r3: the firts PCR to initialize                    */
  106     /* r4: the value to write in the PCRs                 */
  107     /* r5: the number of PCRs to initialize               */
  108     __asm void InitPCRs(void)
  109     {
  110       nofralloc
  111         mtctr r5                         /* intialize ctr with the number of PCRs to initialize */
  112         pcr_init_loop:
  113         sth r4,0(r3)                     /* Write r4 to current PCR address */
  114         addi r3,r3, 2                    /* Inc the memory ptr by 2 to point to the next PCR */
  115         bdnz pcr_init_loop               /* Loop for ctr PCRs */
  116         blr
  117     }                                    /* Initialize the SIU External Bus Interface */
  118     __asm void __initSIUExtBusInterface(void)
  119     {
  120       MAKE_HLI_COMPATIBLE(SIU_PCR0,&SIU.PCR[0].R)
  121         MAKE_HLI_COMPATIBLE(SIU_PCR4,&SIU.PCR[4].R)
  122         MAKE_HLI_COMPATIBLE(SIU_PCR28,&SIU.PCR[28].R)
  123         MAKE_HLI_COMPATIBLE(SIU_PCR62,&SIU.PCR[62].R)
  124         MAKE_HLI_COMPATIBLE(SIU_PCR64,&SIU.PCR[64].R)
  125         MAKE_HLI_COMPATIBLE(SIU_PCR68,&SIU.PCR[68].R)
  126         nofralloc
  127         mflr r27
  128         /*  This initializes the MPC5554 external bus
  129            Set up the pins
  130            Address bus PCR 4 - 27
  131            Configure address bus pins
  132          */
  133         lis r3,SIU_PCR4@h                /* First PCR Address bus is PCR 4 */
  134         ori r3,r3,SIU_PCR4@l
  135         li r5,24                         /* Loop counter to get all address bus PCR (4 to 27) -> 24 PCRs  */
  136         li r4, 0x0440                    /* PCRs initialization value */
  137         bl InitPCRs
  138         /*  Data bus PCR 28-59
  139            Configure data bus pins
  140          */
  141         lis r3,SIU_PCR28@h               /* First PCR for data bus is PCR 28 */
  142         ori r3,r3,SIU_PCR28@l
  143         li r5,32                         /* Loop counter to get all data bus PCR (28-59) -> 32 PCRs  */
  144         li r4, 0x0440                    /* PCRs initialization value */
  145         bl InitPCRs
  146         /*  Configure minimum bus control pins
  147            RD/WR  & BDIP PCR 62/63
  148          */
  149         lis r3,SIU_PCR62@h               /* First PCR for is PCR 62 */
  150         ori r3,r3,SIU_PCR62@l
  151         li r5,2                          /* Loop counter to get all PCR (62-63) -> 2 PCRs  */
  152         li r4, 0x0440                    /* PCRs initialization value */
  153         bl InitPCRs
  154         /*  WE[0-4] PCR 64-67
  155          */
  156         lis r3,SIU_PCR64@h               /* First PCR for is PCR 64 */
  157         ori r3,r3,SIU_PCR64@l
  158         li r5,4                          /* Loop counter to get all PCR (64-67) -> 4 PCRs  */
  159         li r4, 0x0443                    /* PCRs initialization value */
  160         bl InitPCRs
  161         /*  OE & TS
  162          */
  163         lis r3,SIU_PCR68@h               /* First PCR for is PCR 68 */
  164         ori r3,r3,SIU_PCR68@l
  165         li r5,2                          /* Loop counter to get all PCR (68-69) -> 2 PCRs  */
  166         li r4, 0x0443                    /* PCRs initialization value */
  167         bl InitPCRs
  168         /*  Configure the chip selects
  169            CS[0-3]
  170          */
  171         lis r3,SIU_PCR0@h                /* First PCR for is PCR 0 */
  172         ori r3,r3,SIU_PCR0@l
  173         li r5,4                          /* Loop counter to get all PCR (0-3) -> 4 PCRs  */
  174         li r4, 0x0443                    /* PCRs initialization value */
  175         bl InitPCRs
  176         mtlr r27
  177         blr
  178     }                                    /*----------------------------------------------------------------------------*/

  182     /* Initialize the used EBI Chip Selects */
  183     __asm void __initEBIChipSelects(void)
  184     {
  185   
  186   #if INIT_USED_BOARD==MPC5554DEMO_AXM_0321
  187   
  188       MAKE_HLI_COMPATIBLE(EBBI_CS0_BR,&EBI.CS[0].BR.R)
  189         MAKE_HLI_COMPATIBLE(EBBI_CS0_OR,&EBI.CS[0].OR.R)
  190   #endif
  191         nofralloc
  192   #if INIT_USED_BOARD==MPC5554DEMO_AXM_0321
  193   
  194       /*  CY7C1338 512K External SRAM - 4 beat burst, 0 wait
  195          Set up Memory Controller CS0 @ AXM_0321_EXTERNAL_RAM
  196        */
  197       /*  EBI.CS[0].BR.R = (unsigned long)AXM_0321_EXTERNAL_RAM | 0x41UL;
  198        */
  199       lis r3,AXM_0321_EXTERNAL_RAM@h
  200         addi r0,r3,AXM_0321_EXTERNAL_RAM@l
  201         ori r0,r0,0x0041
  202         lis r3,EBBI_CS0_BR@ha
  203         ori r3,r3,EBBI_CS0_BR@l
  204         stw r0,0(r3)
  205       /*  EBI.CS[0].OR.R = 0xfff80000;
  206        */
  207         lis r0,0xfff8
  208         lis r3,EBBI_CS0_OR@h
  209         ori r3,r3,EBBI_CS0_OR@l
  210         stw r0,0(r3)
  211   #endif
  212         blr
  213     }                                    /*----------------------------------------------------------------------------*/
  214     /* Writing to MMU Table Entries                                               */
  215     /*----------------------------------------------------------------------------*/
  216   
  217     /* Write one MMU Table Entry:               */
  218     /* r3, r4, r5 and r6 must hold              */
  219     /* the values of MAS0, MAS1, MAS2 and MAS3  */
  220     __asm void WriteMMUTableEntry( void )
  221     {
  222       nofralloc
  223         /* Write MMU Assist Register 0 (MAS0); SPR 624 */
  224         mtspr 624, r3
  225         /* Write MMU Assist Register 1 (MAS1); SPR 625 */
  226         mtspr 625, r4
  227         /* Write MMU Assist Register 2 (MAS2); SPR 626 */
  228         mtspr 626, r5
  229         /* Write MMU Assist Register 3 (MAS3); SPR 627 */
  230         mtspr 627, r6
  231         /* Write the table entry */
  232         tlbwe
  233         blr
  234     }                                    /* Initialize the needed MMU Table entries */
  235     __asm void __initMMUExternalMemory(void)
  236     {
  237   
  238   #if INIT_USED_BOARD==MPC5554DEMO_AXM_0321
  239   
  240       /* Set up MMU for External Memory
  241          Base address = 0x2000_0000
  242          16 MByte Memory Space, Not Guarded, Cachable, All Access
  243        */
  244       MAKE_HLI_COMPATIBLE(AXM_0321_EXT_RAM_MAS0_VALUE,MAS0_VALUE(2))
  245       /* 16 MB memory space, valid, protected, global which matches with all process IDs */
  246         MAKE_HLI_COMPATIBLE(AXM_0321_EXT_RAM_MAS1_VALUE,MAS1_VALUE(V_VALID,
  247         IPROT_PROTECTED, TID_GLOBAL, 0, TSIZE_16MB))
  248         MAKE_HLI_COMPATIBLE(AXM_0321_EXT_RAM_MAS2_FLAGS,MAS2_FLAGS
  249                             (SHARED_CACHE_STATE_NOT_USED, WRITE_BACK, CACHEABLE,
  250         MEM_COHERENCE_NREQ, NOT_GUARDED, BIG_ENDIAN ))
  251         MAKE_HLI_COMPATIBLE(AXM_0321_EXT_RAM_MAS3_FLAGS,MAS3_FLAGS
  252                             (READ_WRITE_EXECUTE))
  253   #endif
  254         nofralloc
  255         mflr r27
  256   #if INIT_USED_BOARD==MPC5554DEMO_AXM_0321
  257   
  258       /* load r3 with MAS0 value */
  259       lis r3,AXM_0321_EXT_RAM_MAS0_VALUE@h
  260         ori r3,r3,AXM_0321_EXT_RAM_MAS0_VALUE@l
  261         /* load r4 with MAS1 value */
  262         lis r4,AXM_0321_EXT_RAM_MAS1_VALUE@h
  263         ori r4,r4,AXM_0321_EXT_RAM_MAS1_VALUE@l
  264         /* load r5 with the external RAM address from the lcf file */
  265         lis r5,AXM_0321_EXTERNAL_RAM@h
  266         ori r5,r5,AXM_0321_EXTERNAL_RAM@l
  267         /* mask with 0xfffff000 */
  268         clrrwi r5,r5,12
  269         /* copy the masked external ram address to r6 also */
  270         mr r6,r5
  271         /* end MAS2 setup in r3 */
  272         ori r5,r5,AXM_0321_EXT_RAM_MAS2_FLAGS@l
  273         /* end MAS3 setup in r4 */
  274         ori r6,r6,AXM_0321_EXT_RAM_MAS3_FLAGS@l
  275         /* write the MMU entry defined through r3, r4, r5 and r6 */
  276         bl WriteMMUTableEntry
  277   #endif
  278   
  279       mtlr r27
  280         blr
  281     }
