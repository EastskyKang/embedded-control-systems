   24   /* signals options */
   25   #define FS_ETPU_QD_PRIM_SEC            (0)                       /* Primary and secondary only. */
   26   #define FS_ETPU_QD_PRIM_SEC_INDEX      (1)                       /* Index, primary and secondary. */
   27   #define FS_ETPU_QD_PRIM_SEC_HOME       (2)                       /* Home, primary and secondary. */
   28   #define FS_ETPU_QD_PRIM_SEC_INDEX_HOME (3)                       /* Index, Home, primary and secondary. */
   29   
   30   /* configuration options */
   31   #define FS_ETPU_QD_CONFIGURATION_0     (0)                       /* Both QD pins are 0 when the INDEX is reached. */
   32   #define FS_ETPU_QD_CONFIGURATION_1     (1)                       /* Both QD pins are 1 when the INDEX is reached. */
   33   
   34   /* direction options */
   35   #define FS_ETPU_QD_DIRECTION_INC       (0)                       /* Positive Direction. */
   36   #define FS_ETPU_QD_DIRECTION_DEC       (1)                       /* Negative Direction. */
   37   
   38   /* home_transition options */
   39   #define FS_ETPU_QD_HOME_TRANS_LOW_HIGH FS_ETPU_QD_HOME_FM_DETECT_LOW_HIGH /* Detection of low-high transition. */
   40   #define FS_ETPU_QD_HOME_TRANS_HIGH_LOW FS_ETPU_QD_HOME_FM_DETECT_HIGH_LOW /* Detection of high-low transition. */
   41   #define FS_ETPU_QD_HOME_TRANS_ANY      FS_ETPU_QD_HOME_FM_DETECT_ANY /* Detection of any transition. */
   42   
   43   /* index_pulse options */
   44   #define FS_ETPU_QD_INDEX_PULSE_POSITIVE FS_ETPU_QD_INDEX_FM_PULSE_POSITIVE /* Index pulse of positive polarity. */
   45   #define FS_ETPU_QD_INDEX_PULSE_NEGATIVE FS_ETPU_QD_INDEX_FM_PULSE_NEGATIVE /* Index pulse of negative polarity. */
   46   
   47   /* index_pc_reset options */
   48   #define FS_ETPU_QD_INDEX_PC_NO_RESET   FS_ETPU_QD_INDEX_FM_PC_NO_RESET /* Position Counter is NOT reset on Index transition. */
   49   #define FS_ETPU_QD_INDEX_PC_RESET      FS_ETPU_QD_INDEX_FM_PC_RESET /* Position Counter is reset on Index transition. */
   50   
   51   /* etpu_tcr_freq options */
   52   #define FS_ETPU_QD_ETPU_A_TCR1_FREQ    etpu_a_tcr1_freq          /* frequency of eTPU engine A - TCR1 */
   53   #define FS_ETPU_QD_ETPU_A_TCR2_FREQ    etpu_a_tcr2_freq          /* frequency of eTPU engine A - TCR2 */
   54   #define FS_ETPU_QD_ETPU_B_TCR1_FREQ    etpu_b_tcr1_freq          /* frequency of eTPU engine B - TCR1 */
   55   #define FS_ETPU_QD_ETPU_B_TCR2_FREQ    etpu_b_tcr2_freq          /* frequency of eTPU engine B - TCR2 */
   56   
   57   /* mode */
   58   #define FS_ETPU_QD_MODE_SLOW           (1)                       /* Slow mode */
   59   #define FS_ETPU_QD_MODE_NORMAL         (2)                       /* Normal mode */
   60   #define FS_ETPU_QD_MODE_FAST           (4)                       /* Fast mode */
