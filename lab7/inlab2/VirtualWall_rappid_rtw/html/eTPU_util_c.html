   53   /******************************************************************************
   54      FUNCTION     : fs_etpu_init
   55      PURPOSE      : To initialize the eTPU module.
   56      1. Conditionally Load eTPU code into memory.
   57      2. Initialize global registers.
   58      a. MISC vale
   59      b. TCR pre-scalers
   60      3. Copy initial values of global variables to data RAM
   61      INPUTS NOTES : This function has 5 parameters:
   62      p_etpu_config: This is the structure used to initialize the eTPU
   63    *code: This is a pointer to an image of the eTPU code.
   64      code_size: This is the size of the eTPU code in bytes.
   65    *globals: This is a pointer to the global eTPU data that needs
   66      to be initialized.
   67      globals_size: This is the size of the global data in bytes.
   68      RETURNS NOTES: Error code is channel could not be initialized. Error code that
   69      can be returned are:
   70      FS_ETPU_ERROR_CODESIZE: When the code is to big for the
   71      available memory.
   72      WARNING      : This function does not configure the pins only the eTPU->
   73    ******************************************************************************/
   74   uint32_t fs_etpu_init(struct etpu_config_t p_etpu_config, uint32_t *code,
   75                         uint32_t code_size, uint32_t *globals, uint32_t
   76                         globals_size, uint8_t load_microcode)
   77   {
   78     uint32_t *code_end;
   79     int32_t unused_code_ram;
   80     unused_code_ram = (int32_t) (((eTPU->MCR.B.SCMSIZE + 1 ) * 2048) - code_size);
   81     if (unused_code_ram < 0 )
   82       return((uint32_t)FS_ETPU_ERROR_CODESIZE);
   83   
   84     /* 1. Conditionally Load microcode */
   85     if (load_microcode==1) {
   86       /* In order for the MISC function to work with code that is small than
   87          the code memory any unused locations must be set to zero. */
   88       /* enable writting to SCM */
   89       eTPU->ECR_A.B.MDIS = 1;            /* stop eTPU_A */
   90       eTPU->ECR_B.B.MDIS = 1;            /* stop eTPU_B */
   91       eTPU->MCR.B.VIS = 1;               /* enable CPU writes to eTPU code memory */
   92   
   93       /* Copy microcode */
   94       code_end = fs_memcpy32( (uint32_t*)fs_etpu_code_start, code, code_size);
   95   
   96       /* Clear rest of program memory */
   97       fs_memset32( code_end, 0, unused_code_ram);
   98       eTPU->MCR.B.VIS = 0;               /* disable CPU writes to eTPU code memory */
   99     }
  100   
  101     /* Configure MISC */
  102     eTPU->MISCCMPR.R = p_etpu_config.misc;/*write MISC value before enabled in MCR */
  103     eTPU->MCR.R = p_etpu_config.mcr;
  104   
  105     /* Configure Engine 1 */
  106     eTPU->ECR_A.R = p_etpu_config.ecr_a;
  107     eTPU->TBCR_A.R = p_etpu_config.tbcr_a;
  108     eTPU->REDCR_A.R = p_etpu_config.stacr_a;
  109   
  110     /* Configure Engine 2 */
  111     /* Not all parts have a second eTPU engine, if you these write are ignored. */
  112     eTPU->ECR_B.R = p_etpu_config.ecr_b;
  113     eTPU->TBCR_B.R = p_etpu_config.tbcr_b;
  114     eTPU->REDCR_B.R = p_etpu_config.stacr_b;
  115   
  116     /* 3. Copy intial global values to parameter RAM. */
  117     fs_free_param = fs_memcpy32 ( (uint32_t*)fs_etpu_data_ram_start, globals,
  118       globals_size);
  119     return(0);
  120   }

  289   /******************************************************************************
  290      FUNCTION     : fs_etpu_malloc
  291      PURPOSE      : This function allocates data RAM (parameter RAM) for a channel.
  292      INPUTS NOTES : This function has 1 parameter:
  293      num_bytes: this is the number of bytes of data RAM that is
  294      required by a channel. The number of bytes requested
  295      must be a multiple of 8 bytes because the eTPU
  296      alloactes memory in 8 byte blocks.
  297      RETURNS NOTES: A pointer to allocated data RAM. If the requested amount of memory
  298      is larger than the available amount of memory then 0 will be
  299      returned.
  300      WARNING      : This function is non-reentrant and uses the fr_free_param global.
  301    ******************************************************************************/
  302   uint32_t *fs_etpu_malloc(uint16_t num_bytes)
  303   {
  304     uint32_t *pba;
  305   
  306     /*uint32_t x;*/
  307     pba = fs_free_param;
  308     fs_free_param += (((num_bytes+7)>>3)<<1);
  309   
  310     /*    x = (uint32_t) (((num_bytes+7)>>3)<<1);*/
  311     if (fs_free_param > (uint32_t *)fs_etpu_data_ram_end)
  312       return(0);
  313     else
  314       return(pba);
  315   }

  317   /******************************************************************************
  318      FUNCTION     : fs_etpu_malloc2
  319      PURPOSE      : This function is similar to fs_etpu_malloc. The difference is
  320      that this function first checks to see if the CPBA is not sero.
  321      If it is not zero then it assumes the channel has already
  322      been initialized and does not allocate more data RAM to the
  323      channel.
  324      INPUTS NOTES : This function has 2 parameters:
  325      channel: The eTPU channel number.
  326      num_bytes: this is the number of bytes of data RAM that is
  327      required by a channel. The number of bytes requested
  328      must be a multiple of 8 bytes because the eTPU
  329      alloactes memory in 8 byte blocks.
  330      RETURNS NOTES: A pointer to allocated data RAM. If the requested amount of memory
  331      is larger than the available amount of memory then 0 will be
  332      returned.
  333      WARNING      : This function is non-reentrant and uses the fr_free_param global.
  334    ******************************************************************************/
  335   uint32_t *fs_etpu_malloc2(uint8_t channel, uint16_t num_bytes)
  336   {
  337     uint32_t *pba;
  338     if (eTPU->CHAN[channel].CR.B.CPBA == 0 ) {
  339       pba = fs_free_param;
  340       fs_free_param += (((num_bytes+7)>>3)<<1);
  341       if (fs_free_param > (uint32_t *)fs_etpu_data_ram_end)
  342         return(0);
  343       else
  344         return(pba);
  345     } else
  346       return(fs_etpu_data_ram(channel));
  347   }

  655   /******************************************************************************
  656      FUNCTION     : fs_memcpy32
  657      PURPOSE      : This function is similar to the standard C memcpy() function
  658      however it copies 32 bit words rather than bytes.
  659      INPUTS NOTES : This function has 3 parameters:
  660    *dest: This is a pointer to the destination location
  661    *source this is a pointer to the source location
  662      size: This is the size of the data to copy in bytes.
  663      RETURNS NOTES: A pointer to the end of the copied data.
  664      WARNING      : The *dest and *source pointers should be aligned to a 32 bit
  665      address. If they are not it may cause memory exceptions. Moving
  666      data to the eTPU code memory that is not 32 bit aligned is
  667      undefined. The size should be a multiple of 4 to copy 32 bit
  668      value. If it is not it is rounded down.
  669    ******************************************************************************/
  670   uint32_t *fs_memcpy32(uint32_t *dest, uint32_t *source, uint32_t size)
  671   {
  672     uint32_t *p = dest;
  673     uint32_t *q = source;
  674     size = size >>2;
  675     while (size--)
  676       *p++ = *q++;
  677     return (p);
  678   }

  680   /******************************************************************************
  681      FUNCTION     : fs_memset32
  682      PURPOSE      : This function is similar to the standard C memset() function
  683      however it sets 32 bit words rather than bytes.
  684      INPUTS NOTES : This function has 3 parameters:
  685    *start: This is a pointer to the start location
  686      value: This is the value to write to memory
  687      size: This is the size of the data to copy in bytes.
  688      RETURNS NOTES: none.
  689      WARNING      : The *start pointer should be aligned to a 32 bit address. If
  690      it is not it may cause memory exceptions. Writing data to the
  691      eTPU code memory that is not 32 bit aligned is undefined. The
  692      size should be a multiple of 4 to copy 32 bit  value. If it
  693      is not it is rounded down.
  694    ******************************************************************************/
  695   void fs_memset32(uint32_t *start, uint32_t value, int32_t size)
  696   {
  697     uint32_t *p = start;
  698     size = size >>2;
  699     while (size--)
  700       *p++ = value;
  701   }
